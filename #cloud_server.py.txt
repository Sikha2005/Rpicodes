#cloud_server.py

from flask import Flask, request, jsonify

app = Flask(__name__)

@app.route("/cloud", methods=["POST"])
def receive_data():
    data = request.get_json()
    print(" Received from RPi:", data)
    return jsonify({"status": "OK"}), 200

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=6000)
